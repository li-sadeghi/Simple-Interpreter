var result:int = 0;

func fibonacci(n:int): int {
  if (n <= 1) {
    return n;
  } else {
    return fibonacci(n - 1) + fibonacci(n - 2);
  }
}


func sum_fibonacci(limit:int): int {
  var total:int = 0;
  var i:int = 0;
  while (i <= limit) {
    total = total + fibonacci(i);
    i = i + 1;
  }
  return total;
}

result = sum_fibonacci(10);

print(result);
